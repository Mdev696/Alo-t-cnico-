-- 1. LIMPEZA TOTAL (Isso apagará os dados antigos para recriar com as colunas certas)
drop table if exists registros_alerta;

-- 2. CRIAÇÃO DA TABELA (Colunas batendo com o seu JavaScript)
create table registros_alerta (
  id uuid default gen_random_uuid () primary key,
  ticket text,
  tecnico_nome text,
  empresa text, -- Alterado de 'equipe' para 'empresa' para bater com o HTML
  setor text,
  data_iso date,
  endereco text,
  tipo text,
  informacoes text,
  qtd_fotos int,
  fotos_urls text,
  status text default 'Pendente',
  created_at timestamp with time zone default timezone ('utc'::text, now())
);

-- 3. HABILITAR SEGURANÇA (RLS)
alter table registros_alerta enable row level security;

-- 4. POLÍTICA DE ACESSO (Liberado para usuários logados)
create policy "Acesso Total Autenticado" on registros_alerta for all using (auth.role () = 'authenticated');

-- 5. ATUALIZAÇÃO DO CACHE DO SISTEMA
notify pgrst,
'reload schema';
